389  //Если сдвиг вниз.

391  {
392  //Перемещение данных
393  for (int i = (arrRowCount - 1); i >= offsetFirstRow; i--)

395 
396 
397  //Очистка массива
398  for (int i = offsetFirstRow; i < (offsetFirstRow + offsetRowCount); i++)
399  for (int j = 0; j < colCount; j++) arr[i, j] = "";
400

402  }
403 
404  //Если сдвиг вверх (т.е. удаление строк).
405  if (offsetRowCount < 0)

407  offsetRowCount = offsetRowCount * -1;
408 
409  //Перемещение данных

411  for (int j = 0; j < colCount; j++) arr[i - offsetRowCount, j] = arr[i, j];
