110  public bool LoadEntityTree()
111  {
112  if (Var.IsDesignMode) return false;
113  treeViewEntity.ImageList = imageList1;
114  //sys.LoadTreeFromDataTable(treeViewAttr, DTEnt, "ID", "ParentID", "Name", true);
115  //TreeNode argentinaNode = new TreeNode { Text = "Аргентина", ImageIndex=0, SelectedImageIndex=0 };
116  //treeView1.Nodes.Add(argentinaNode);
117  if (!sys.SelectDT(DirectionQuery.Remote, "SELECT * FROM fbaEntity ", out dtEnt)) return false; //WHERE ID = 1694
118  treeViewEntity.Nodes.Clear();
119  foreach (DataRow dr in dtEnt.Select("ParentID = 0 OR ParentID IS NULL"))
120  {
121  var node = new TreeNode();
122  node.Text = dr["Name"].ToString();
123  node.Tag = dr["ID"];
124  string Feature = dr["Feature"].ToString();
125 
126 
127  int numberInt;
128  int imageIndex = 2;
129  if (sys.IsInteger(Feature, out numberInt))
130  {
131  if (sys.GetBit(numberInt, 0)) imageIndex = 0;
132  if (sys.GetBit(numberInt, 1)) imageIndex = 1;
133  }
134  node.ImageIndex = imageIndex;
135  node.SelectedImageIndex = node.ImageIndex;
136  treeViewEntity.Nodes.Add(node);
137  AddNodes(dtEnt, node);
138  }
139  treeViewEntity.Sort();
140  return true;
141  }

148  private void AddNodes(System.Data.DataTable dt, TreeNode node)
149  {
150  foreach (DataRow dr in dt.Select("ParentID = " + node.Tag.ToString()))
151  {
152  var node1 = new TreeNode();
153  node1.Text = (string)dr["Name"]; //1- синий, 2 - зеленый, 3 серый
154  node1.Tag = dr["ID"];
155  string Feature = dr["Feature"].ToString();
156  if (Feature == "1") node1.ImageIndex = 0;
157  else if (Feature == "5") node1.ImageIndex = 0;
158  else if (Feature == "2") node1.ImageIndex = 2;
159  else if (Feature == "4") node1.ImageIndex = 2;
160  else if (Feature == "34") node1.ImageIndex = 2;
161  else node1.ImageIndex = 1;
162  node1.SelectedImageIndex = node1.ImageIndex;
163  node.Nodes.Add(node1);
164  AddNodes(dt, node1);
165  }
