691  private bool GetLookupSQL(out string sql)
692  {
693  sql = "";
694  if (!sys.IsEmpty(СustomQuery)) sql = СustomQuery;
695  else if (ObjRef != null) sql = ObjRef.GetSelectSQLForSysDBEdit(this);
696  else
697  {
698  if ((!sys.IsEmpty(EntityBrief)) && (!sys.IsEmpty(AttrBrief)))
699  {
700  string queryName;
701  string attr;
702  string attrLookup;
703  string setting;
704  sys.ParseTag(AttrBrief, out queryName, out attr, out attrLookup, out setting);
705  string maxRecords1 = "";
706  string maxRecords2 = "";
707  if (ObjectMaxCount > 0)
708  {
709  if (Var.con.serverTypeRemote == ServerType.MSSQL)
710  maxRecords1 = " TOP " + ObjectMaxCount.ToString();
711  if ((Var.con.serverTypeRemote == ServerType.SQLite) || (Var.con.serverTypeRemote == ServerType.Postgre))
712  maxRecords2 = Var.CR + " LIMIT " + ObjectMaxCount.ToString();
713  }
714  sql = "SELECT " + maxRecords1 + " " + attrLookup + " FROM " + EntityBrief;
715  if (!sys.IsEmpty(OuterWHERE)) sql = sql + " WHERE " + OuterWHERE;
716  sql = sql + maxRecords2;
717  if (!sys.IsEmpty(ObjectOrderBy)) sql = sql + " ORDER BY " + ObjectOrderBy;
718  }
719  }
720  if (sql == "") return false;
721  return true;
722  }
