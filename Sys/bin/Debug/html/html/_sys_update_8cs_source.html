79  public static bool UpdateCheck(out string version,
80  out string numberUpdate,
81  out string resultMessage,
82  bool showMes)
83  {
84  version = "";
85  numberUpdate = "";
86  resultMessage = "";
87  try
88  {
89  var sql = "";
90  if (Var.con.serverTypeRemote == ServerType.MSSQL)
91  sql = "SELECT TOP 1 Version, NumberUpdate FROM fbaUpdate " +
92  "WHERE UserUpdate = 1 AND Version = (SELECT TOP 1 CurrentVersion FROM fbaUpdate);";
93  else
94  if ((Var.con.serverTypeRemote == ServerType.SQLite) || (Var.con.serverTypeRemote == ServerType.Postgre))
95  sql = "SELECT Version, NumberUpdate FROM fbaUpdate " +
96  "WHERE UserUpdate = 1 AND Version = (SELECT CurrentVersion FROM fbaUpdate LIMIT 1) LIMIT 1;";
97  if (sql == "") return false;
98  sys.GetValue(DirectionQuery.Remote, sql, out version, out numberUpdate);
99  if (version == "")
100  {
101  resultMessage = "Ошибка при проверке обновления: Нет информации об актуальной версии приложения!";
102  if (showMes) sys.SM(resultMessage, MessageType.Information);
103  return false;
104  }
105  bool needUpdate = version != Var.ApplicationVersion;
106  if (needUpdate)
107  resultMessage = "Требуется обновление." + Var.CR + "Новая версия программы: " + version + Var.CR +
108  "Номер обновления: " + numberUpdate + Var.CR + "Текущая версия программы: " +
109  Var.ApplicationVersion;
110  else resultMessage = "Обновление не требуется. У Вас уже установлена самая последня версия: " + version + ", номер обновления: " + numberUpdate;
111  if (showMes) sys.SM(resultMessage, MessageType.Information);
112  return needUpdate;
113  }
114  catch (Exception ex)
115  {
116  resultMessage = "Ошибка при проверке обновления: " + ex.Message;
117  if (showMes) sys.SM(resultMessage, MessageType.Information);
118  return false;
119  }
120  }

148  public static bool UpdateDownload(string version, string numberUpdate, out string resultMessage, out bool needUpdate, bool showMes)
149  {
150  resultMessage = "";
151  DataTable dT1;
152  DataTable dT2;
153  string sqlLocal;
154  long fileSize = 0;
155  var fileCount = 0;
156  var listDelete = new List<string>();
157  string[] listNotDeleted;
158  //Очищаем полность папку sys.PathUpdate.
159  if (!FBAFile.DirClean(FBAPath.PathUpdate, out listNotDeleted, true))
160  {
161  needUpdate = false;
162  return false;
163  }
164 
165  sqlLocal = "select ID, MD5, Path, FullName, ContentType, Operation, Size from fbaUpdate WHERE Version = '" + version + "' ORDER BY Numberupdate, Numberfile; ";
166  sys.SelectDT(DirectionQuery.Remote, sqlLocal, out dT1);
167  if (dT1.Rows.Count == 0)
168  {
169  resultMessage = ""; //Ошибка обновления. Не удается найти последниюю версию приложения!";
170  needUpdate = false;
171  return false;
172  }
173 
174  var progress = new FormProgress("Обновление программы", "Получение файлов для обновления", dT1.Rows.Count);
175  if (showMes) progress.Show();
176  for (var i = 0; i < dT1.Rows.Count; i++)
177  {
178  var id = dT1.Value(i, "ID");
179  var md5Update = dT1.Value(i, "MD5");
180  var pathValue = dT1.Value(i, "Path");
181  var fullName = dT1.Value(i, "FullName");
182  var contentType = dT1.Value(i, "ContentType");
183  var operationL = dT1.Value(i, "Operation");
184  var size = dT1.Value(i, "Size");
185  var fileNameLocal = fullName.Replace(pathValue, FBAPath.PathMain);
186  var FileNameUpdate = fullName.Replace(pathValue, FBAPath.PathUpdate);
187  if (operationL == "DELFILE" || operationL == "DELDIR") listDelete.Add(operationL + ": " + fullName);
188  if (operationL == "ADDDIR") Directory.CreateDirectory(Path.GetDirectoryName(FileNameUpdate));
189  if (operationL == "ADDFILE")
190  {
191  var MD5Local = Crypto.FileHashMD5Calc(fileNameLocal); //MD5 локального файла.
192  var MD5Download = Crypto.FileHashMD5Calc(FileNameUpdate); //MD5 уже скаченного ранее файла.
193 
194  //Если рабочий файл не совпадает с обновлением:
195  if (md5Update != MD5Local)
196  {
197  fileSize = fileSize + Convert.ToInt64(size);
198  fileCount = fileCount + 1;
199 
200  //Если MD5 не совпадает между локальным файлом и файлом для обновления, то скачиваем файл.
201  sqlLocal = "SELECT FileData FROM fbaUpdate WHERE ID = " + id;
202  sys.SelectDT(DirectionQuery.Remote, sqlLocal, out dT2);
203  var FileData = dT2.Value("FileData");
204  Directory.CreateDirectory(Path.GetDirectoryName(FileNameUpdate));
205  if (!FBAFile.FileWriteFromBase64(FileData, FileNameUpdate, out resultMessage, showMes))
206  {
207  resultMessage = "Ошибка обновления. Не удается найти последниюю версию приложения!" +
208  Var.CR + resultMessage;
209  needUpdate = false;
210  return false;
211  }
212  }
213  }
214  if (listDelete.Count > 0)
215  {
216  //Просто сохраняем в список то, что нужно удалить.
217  var fileName = FBAPath.PathUpdate + "ListDelete.txt";
218  var listStr = string.Join(Var.CR, listDelete.ToArray());
219  File.WriteAllText(fileName, listStr, Encoding.Default);
220  }
221  if (showMes) progress.Inc();
222  }
223 
224  progress.Dispose();
225  if (fileCount == 0 && listDelete.Count == 0)
226  {
227  resultMessage = "Обновление не требуется. Все текущие версии файлов совпадают с файлами обновления " +
228  version;
229  needUpdate = false;
230  return true;
231  }
232 
233  //Если нужно только удалить папки или файлы, то ставим размер обновления 1 кб,
234  //чтобы не пугать пользователя.
235  if (fileSize == 0) fileSize = 1024;
236  var ResStr = "Порядковый номер обновления: " + numberUpdate + Var.CR + "Версия: " + version + Var.CR +
237  "Размер обновления: " + FBAFile.GetFileSizeStr(fileSize, true, true) + Var.CR +
238  "Количество файлов и папок: " + (fileCount + listDelete.Count) + Var.CR +
239  "Текущая версия программы: " + Var.ApplicationVersion;
240  resultMessage = "Файлы для обновления скачаны и готовы для установки." + Var.CR + ResStr;
241  needUpdate = true;
242  return true; //true - обновлять нужно, false - обновление не требуется.
243  }
