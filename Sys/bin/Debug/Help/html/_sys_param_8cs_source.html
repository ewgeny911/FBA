217  public static string GetComponentValues(Control.ControlCollection controls)
218  {
219  if (controls.Count == 0) return "";
220  string settingText = "";
221  foreach(Control control in controls)
222  {
223  settingText = settingText + GetComponentValues(control.Controls);
224  if (control.Tag != null)
225  {
226  //Пример формата строки в Tag: SAVE
227  string compType = control.GetType().ToString();
228  compType = sys.TruncateType(compType, false);
229  string setting = control.Tag.ToString().ToUpper();
230  if (setting.IndexOfEx("SAVE") > -1) setting = "SAVE";
231  bool saveParam = false;
232  if (compType == "TextBoxFBA") saveParam = ((TextBoxFBA)control).SaveParam;
233  else if (compType == "ComboBoxFBA") saveParam = ((ComboBoxFBA)control).SaveParam;
234  else if (compType == "CheckBoxFBA") saveParam = ((CheckBoxFBA)control).SaveParam;
235  else if (compType == "RadioButtonFBA") saveParam = ((RadioButtonFBA)control).SaveParam;
236  else if (compType == "TabControlFBA") saveParam = ((RadioButtonFBA)control).SaveParam;
237  if (saveParam) setting = "SAVE";
238 
239  if (setting == "SAVE")
240  {
241  string settingStr = "";
242 
243  if ((compType == "TextBox") || (compType == "TextBoxFBA"))
244  {
245  settingStr = control.Name + ":" + control.Text.StringToBase64() + Var.CR;
246  } else
247  if ((compType == "CheckBox") || (compType == "CheckBoxFBA"))
248  {
249  settingStr = control.Name + ":" + ((System.Windows.Forms.CheckBox)control).Checked + Var.CR;
250  } else
251  if ((compType == "ComboBox") || (compType == "ComboBoxFBA"))
252  {
253  settingStr = control.Name + ":" + control.Text.StringToBase64() + Var.CR;
254  } else
255  if ((compType == "RadioButton") || (compType == "RadioButtonFBA"))
256  {
257  settingStr = control.Name + ":" + ((System.Windows.Forms.RadioButton)control).Checked + Var.CR;
258  } else
259  if ((compType == "TabControl") || (compType == "TabControlFBA"))
260  {
261  settingStr = control.Name + ":" + ((System.Windows.Forms.TabControl)control).SelectedIndex + Var.CR;
262  } else
263  settingStr = control.Name + ":" + control.Text.StringToBase64() + Var.CR;
264 
265  settingText += settingStr;
266  }
267  }
268  }
269  return settingText;

278  public static bool SetComponentValues(string componentValues, Control.ControlCollection controls)
279  {
280  if (componentValues == "") return false;
281  string[] dotArray = componentValues.Split('\n');
282  for (int i = 0; i < dotArray.Count(); i++)
283  {
284  string s = dotArray[i].Trim();
285  int n = s.IndexOf(':');
286  if (n == -1) continue;
287  string controlName = s.Substring(0, n);
288  string controlValue = s.Substring(n + 1);
289  Control control = controls.Find(controlName, true).FirstOrDefault();
290  if (control != null)
291  {
292  string compType = control.GetType().ToString();
293  compType = compType.Replace("System.Windows.Forms.", "");
294  compType = compType.Replace("FBA.", "");
295  compType = compType.Replace("FBA", "");
296 
297  if (compType == "TextBox")
298  {
299  control.Text = controlValue.Base64ToString();
300  } else
301  if (compType == "CheckBox")
302  {
303  ((System.Windows.Forms.CheckBox)control).Checked = controlValue.ToBool();
304  } else
305  if (compType == "ComboBox")
306  {
307  control.Text = controlValue.Base64ToString();
308  } else
309  if (compType == "RadioButton")
310  {
311  ((System.Windows.Forms.CheckBox)control).Checked = controlValue.ToBool();
312  } else
313  if (compType == "TabControl")
314  {
315  ((System.Windows.Forms.TabControl)control).SelectedIndex = controlValue.ToInt();
316  } else
317  control.Text = controlValue.Base64ToString(); //При этом присвоении вызовется ControlValueChanged автоматически.
318  }
319  }
320  return true;
321  }
