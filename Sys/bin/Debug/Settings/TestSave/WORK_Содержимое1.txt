SELECT top 1
       ЛПУ LPUID,
       ЛПУ.СпособОтправки.Наим SendingMethod,
       ЛПУ.СпособОтправки.ИДОбъекта SendingMethodID,
       ЛПУ.ЮЛ.EMail EMail, ЛПУ.ЮЛ.АдресПочтовый MailAddress, ЛПУ.ЮЛ.Факс Fax,
       ПрограммаВариантДогСтрах.СтрахПрог.Сокр ProgramBrief,
       ПрограммаВариантДогСтрах.СтрахПрог.Наим ProgramName,
       case when ПрограммаВариантДогСтрах.Прикрепление = 1 then 'Предоплата'
            else 'Факт' end PayFact,
       ЛПУ.Код LPUCode,
       ЛПУ.Наим LPUName,
       Null ResShow
FROM СодержимоеВариантДогСтрах
WHERE ВариантДогСтрах.ДогСтрах.Номер = '77050102-00204' -- = 3325151
  AND (ДатаПрикрепления <= '20180810 00:00:00')
  AND (ДатаОткрепления is null OR ДатаОткрепления >= '20180810 00:00:00')
  AND ПрограммаВариантДогСтрах.СтрахПрог.СпискиЛПУ = 1
