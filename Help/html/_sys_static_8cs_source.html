340  public static bool SendErrorToServer(string errorMes, string additionalInfo)
341  {
342  //Выходим, если делать ничего не нужно.
343  if (!SaveError) return false;
344 
345  //Это для того чтобы если во время работы этого кода возникнет ошибка, не было зацикливания.
346  SaveError = false;
347 
348  try
349  {
350  //Если чтение параметра неудачное, то ничего не делаем.
351  var Params = new string[10];
352 
353  //Получаем только при первой ошибке.
354  if (ErrorsCount == 0)
355  {
356  if (Param.Load(DirectionQuery.Remote, Var.UserID, "SaveError", true, "User", out Params))
357  {
358  if (!Params[0].Replace("SaveError=", "").ToBool()) return false;
359  SaveScreenshot = Params[1].Replace("SaveScreenshot=", "").ToBool();
360  CompressRatio = Params[2].Replace("CompressRatio=", "").ToInt();
361  SecBetweenSendError = Params[3].Replace("SecBetweenSendError=", "").ToInt();
362  }
363  }
364 
365  //sys.Error.SaveError true/false: Сохранять ошибку на сервере или нет. Пример: SaveError=true
366  //sys.Error.SaveScreenshot true/false: Сохранять текст ошибки на сервере или нет. Пример: SaveScreenshot=true
367  //sys.Error.CompressRatio int: находится значение степени сжатия скриншота. По умолчанию формат JPEG, степень сжатия 90. Пример: CompressRatio=90
368  //sys.Error.SecBetweenSendError int: - количество секунд между ошибками.
369  //sys.Error.LastDateTimeSendError datetime: Время последней ошибки.
370 
371  if (sys.GetSecDiff(LastDateTimeSendError, DateTime.Now) < SecBetweenSendError)
372  {
373  SaveError = true;
374  return false;
375  }
376 
377  string imagebase64 = "";
378  string screenshoFormat = "";
379  string screenshotWidth = "";
380  string screenshotHeight = "";
381  string screenshotSize = "";
382 
383  if (SaveScreenshot)
384  {
385  //Получение скриншота
386  screenshoFormat = "WEBP";
387  screenshotWidth = Screen.PrimaryScreen.Bounds.Width.ToString();
388  screenshotHeight = Screen.PrimaryScreen.Bounds.Height.ToString();
389  Bitmap printscreen = new Bitmap(Screen.PrimaryScreen.Bounds.Width, Screen.PrimaryScreen.Bounds.Height);
390  Graphics graphics = Graphics.FromImage(printscreen as Image);
391  graphics.CopyFromScreen(0, 0, 0, 0, printscreen.Size);
392  byte[] webpImageData;
393  WebP.EncodeLossly(printscreen, CompressRatio, out webpImageData);
394  imagebase64 = Convert.ToBase64String(webpImageData);
395  screenshotSize = FBAFile.GetFileSizeStr(webpImageData.Length, true, false);
396  }
397  additionalInfo = sys.AddRightCR(additionalInfo);
398  additionalInfo += GetSystemInfo();
399  string sql = "INSERT INTO fbaError (EntityID, UserID, ErrorTime, ErrorText, ScreenshotFormat, ScreenshotWidth, ScreenshotHeight, ScreenshotSize, " +
400  "CompressRatio, AdditionalInfo, ScreenshotData) VALUES (" + Var.CR +
401  "123, " + Var.UserID + "," + sys.DateTimeCurrent() + ",'" + errorMes + "','" + screenshoFormat + "','" + screenshotWidth + "','" + screenshotHeight + "'," + screenshotSize + ",'" + Error.CompressRatio.ToString() + "','" + additionalInfo + "','" + imagebase64 + "')";
402 
403  sys.Exec(DirectionQuery.Remote, sql);
404  LastDateTimeSendError = DateTime.Now;
405  }
406  catch
407  {
408  //Ошибку не выдаем.
409  }
410 
411  SaveError = true;
412  return true;
413  }
